\begin{case}不存在奇解  \\
    $ f_y(x, y)$存在且有界，
    也就是解的存在唯一性定理在定义区间上全部满足
\end{case}

\begin{case} 包络线及奇解的求法\\
    是包络线一定是奇解\\
    \begin{case} 
        C-判别
        $$
        \left\{
            \begin{array}{l}
        \Phi(x, y, c)=0 \\
        \Phi c^{\prime}(x, y, c)=0
        \end{array}
        \right.
        $$\\    
    \end{case}
    \begin{case} 
        p-判别  
        $$
        \left\{
            \begin{array}{l}
        F(x, y, p)=0 \\
        F c^{\prime}(x, y, p)=0
        \end{array}
        \right.
        $$\\
    \end{case}
\end{case}
